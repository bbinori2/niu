            const content3Items = document.querySelectorAll('#content3 .item');
            const content3Pages = document.querySelectorAll('#content3 .page');
            const itemsPerPage = 1;

            function displayItems(pageToShow, items, pages) {
                const startIndex = (pageToShow - 1) * itemsPerPage;
                const endIndex = startIndex + itemsPerPage;

                items.forEach((item, index) => {
                    if (index >= startIndex && index < endIndex) {
                        item.style.display = 'block';
                    } else {
                        item.style.display = 'none';
                    }
                });
            }

            displayItems(1, content1Items, content1Pages);
            displayItems(1, content2Items, content2Pages);
            displayItems(1, content3Items, content3Pages);

            function handlePagination(pages, items) {
                pages.forEach((page, index) => {
                    page.addEventListener('click', () => {
                        pages.forEach((p) => p.classList.remove('active'));
                        page.classList.add('active');
                        displayItems(index + 1, items, pages);
                    });
                });
            }

            handlePagination(content1Pages, content1Items);
            handlePagination(content2Pages, content2Items);
            handlePagination(content3Pages, content3Items);
